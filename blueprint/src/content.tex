\section{Prefix codes}

Let $\{0,1\}$ be a set. Recall that a word over $\{0,1\}$ is a sequence of elements of $\{0,1\}$, the set of all words over $\{0,1\}$
including the empty word is denoted by $\{0,1\}^\ast$. We will write $xy$ for the concatenation of words $x, y\in \{0,1\}^\ast$. We denote the empty word by \(\varepsilon\).
We denote the length of a word \(x\in \{0,1\}^*\) by \(|x|\).
A word $x$ is a \emph{prefix} of a word $y$ if there exists a word $z$ such that $xz = y$. We write $x \preceq y$
to denote that $x$ is a prefix of $y$.
\begin{definition}%
  \label{def:comparable}
  \rocq{Thompson.codes.comparable}
  \rocqok{}
  Let $x, y\in A^\ast$. We say that $x$ and $y$ are \emph{comparable} if $x$ is a prefix of $y$ or vice-versa, i.e.
  $x \preceq y$ or $y \preceq x$ holds. If $x$ and $y$ are not comparable, we say they are \emph{incomparable}.
\end{definition}

\begin{lemma}%
  \label{lem:comparable_symm}
  \uses{def:comparable}
  \rocq{Thompson.codes.comparable_symm}
  \rocqok{}
  Let $x, y\in A^\ast$. Then $x$ and $y$ are comparable if and only if $y$ and $x$ are comparable.
\end{lemma}
\begin{proof}
  \rocqok{}
  The proof follows immediately from the definition, as ``or'' is symmetric.
\end{proof}

An important role in the theory of Thompsons groups is played by prefix codes, which we now define.

\begin{definition}%
  \label{def:prefix_code}
  \uses{def:comparable}
  \rocq{Thompson.codes.prefix_code}
  \rocqok{}
  A prefix code over $\{0,1\}$ is a set of words $P$ over $\{0,1\}$ such that every pair of distinct
  words $w_1, w_2\in P$ are incomparable.
\end{definition}



\begin{definition}\label{def:concat_letter_set}
    Let \(x\in \{0,1\}\) and \(S\subseteq \{0,1\}^*\) be finite. We denote the set \(\makeset{xs}{\(s\in S\)}\) by \(xS\).
\end{definition}

\section{The Cantor set}
The Thompson's groups act faithfully on the Cantor space of all infinite binary sequences. 
This perspective makes some proofs easier so we will go back and forth between the two perspectives.
\begin{definition}\label{def:Cantor_space}

  This set is viewed as a topological space with the product topology using the discrete topology on \(\{0,1\}\).
  We refer to this topological space as the \emph{Cantor space} and denote it by \(\mathfrak{C}\).

\begin{definition}\label{def:concat_fin_inf}
\uses{def:Cantor_space}
    Suppose that \(w\in \{0,1\}^*\) and \(x\in \{0,1\}^{\mathbb{N}}\). We write \(wx\) to denote the map from \(\mathbb{N}\) to \(\{0,1\}\) defined by
    \[(n)wx=\begin{cases}
        (n)w & \text{ if }n<|w|\\
        (n-|w|)x & \text{ if }n\geq |w|.        
    \end{cases}.\]
    More generally. If \(S\subseteq \mathfrak{C}\), then we write \(wS\) for the set \(\makeset{wx}{\(x\in S\)}\).
\end{definition}

\begin{definition}\label{def:cones}
\uses{def:concat_fin_inf}
Suppose that \(w\in \{0,1\}^*\). We define the \emph{cone} of \(w\) to be the set
\(w\mathfrak{C}\).
\end{definition}

Recall that for \(v,w\in \{0,1\}^*\), we write \(w\preceq v\) if \(v\in w\{0,1\}^*\).
Hence the following definition.
\begin{definition}\label{def:finite_prefix_of_cantor}
\uses{def:Cantor_space}
We say that \(w\in \{0,1\}^*\) is a \emph{prefix} of \(x\in \{0,1\}^{\mathbb{N}}\) if \(x\in w\mathfrak{C}\). We write $w \preceq x$
to denote that $w$ is a prefix of $x$.
\end{definition}

\begin{definition}\label{def:restriction_of_infinite_string}
\uses{def:Cantor_space}
If \(x\in \mathfrak{C}\) and \(n\in \mathbb{N}\), then we write \(x|_n\) for the map with domain \(\{0,1, \ldots, n-1\}\) with \((i)x|_n=(i)x\) for all \(i<n\).
\end{definition}
\begin{definition}\label{def:restriction_of_finite_string}
If \(v\in \{0,1\}^*\) and \(n\in \mathbb{N}\) satisfies \(n\leq |v|\), then we write \(v|_n\) for the map with domain \(\{0,1, \ldots, n-1\}\) with \((i)v|_n=(i)v\) for all \(i<n\).
\end{definition}

The following description of cones is often more helpful than the definition.
\begin{lemma}\label{lem:cone_prefixes}
\uses{def:finite_prefix_of_cantor,def:cones,def:restriction_of_infinite_string}
    Suppose that \(w\in \{0,1\}^*\) and \(x\in \mathfrak{C}\). Then \(x\in w\mathfrak{C}\) if and only if \(w=x|_{|w|}\).
\end{lemma}
\begin{proof}
    \((\Leftarrow):\) Suppose that \(w =x|_{|w|}\). Define \(x':\mathbb{N}\to \{0,1\}\) by \((n)x'=(n+|w|)x\).
    As \(wx'\in w\mathfrak{C}\), we need only show that \(wx'=x\). Let \(n\in \mathbb{N}\) be arbitrary. We need only show that \((n)wx'=x\). If \(n<|w|\), then we have \((n)wx'=(n)w=(n)x|_{|w|}=(n)x\) as required.
    If \(n\geq |w|\), then \((n)wx'=(n-|w|)x'=(n-|w|+|w|)x=(n)x\) as required.

    \((\Rightarrow):\) Suppose that \(x\in w\mathfrak{C}\). It follows that there is \(x'\in \mathfrak{C}\) such that \(x=wx'\). In particular, for all \(n<|w|\), we have \((n)x|_{|w|}=(n)x=(n)wx'=(n)w\) as required.
\end{proof}
The same works for finite prefixes.
\begin{lemma}\label{lem:finite_cone_prefixes}
\uses{def:finite_prefix_of_cantor,def:cones, def:restriction_of_finite_string}
    Suppose that \(w,v\in \{0,1\}^*\). Then \(v\in w\{0,1\}^*\) if and only if \(|v|\geq |w|\) and \(w=v|_{|w|}\).
\end{lemma}
\begin{proof}
    \((\Leftarrow):\) Suppose that \(|v|\geq |w|\) and \(w =v|_{|w|}\). Define \(v':\{0,1 ,\ldots |v|-|w|-1\}\to \{0,1\}\) by \((n)v'=(n+|w|)v\).
    As \(wv'\in w\{0,1\}^*\), we need only show that \(wv'=v\). First note that \(|v|=|w|+|v'|=|w|+|v|-|w|=|w|\).
    Let \(n<|v|\) be arbitrary. We need only show that \((n)wv'=v\). If \(n<|w|\), then we have \((n)wv'=(n)w=(n)v|_{|w|}=(n)v\) as required.
    If \(n\geq |w|\), then \((n)wv'=(n-|w|)v'=(n-|w|+|w|)v=(n)v\) as required.

    \((\Rightarrow):\) Suppose that \(v\in w\{0,1\}^*\). It follows that there is \(v'\in \{0,1\}^*\) such that \(v=wv'\).
    We have \(|v|=|w|+|v'|\geq |w|\).
    Moreover, for all \(n<|w|\), we have \((n)v|_{|w|}=(n)v=(n)wv'=(n)w\) as required.
\end{proof}

\begin{lemma}\label{lem:double_restrict}
\uses{lem:cone_prefixes,lem:finite_cone_prefixes}
    Suppose that \(v,w\in \{0,1\}^*\) and \(x\in \mathfrak{C}\) are such that \(w\preceq v\) and \(v\preceq x\). Then \(w\preceq x\).
\end{lemma}
\begin{proof}
By the definition of \(\preceq\) and Lemma~\ref{lem:finite_cone_prefixes}, it follows that \(v|_{|w|}=w\). Similarly, using Lemma~\ref{lem:cone_prefixes}, we have \(x|_{|v|}=v\).
By Lemma~\ref{lem:cone_prefixes} we need to show that \(x|_{|w|}=w\).
This follows as for all \(n<|w|\) we have \((n)x|_{|w|}=(n)x=(n)x|_{|v|}=(n)v=(n)v|_{|w|}=(n)w\).
\end{proof}

\begin{lemma}\label{lem:comp_cone_vs_word}
\uses{def:cones,lem:cone_prefixes,lem:double_restrict}
    If \(v,w\in \{0,1\}^*\), then \(w\preceq v \iff w\mathfrak{C} \supseteq v\mathfrak{C}\).
\end{lemma}
\begin{proof}
\((\Rightarrow):\) Suppose that \(w\preceq v\) and \(x\in \mathfrak{C}\). We show that \(vx\in w\mathfrak{C}\). In other words, we must show that \(w\preceq vx\). As \(w\preceq v\) and \(v \preceq vx\), this follows from Lemma~\ref{lem:double_restrict}.

\((\Leftarrow):\) Suppose that \(v\mathfrak{C} \subseteq w\mathfrak{C}\). We show that \(w\preceq v\). 
Suppose for a contradiction that \(w\) is not a prefix of \(v\). It follows from Lemma~\ref{lem:finite_cone_prefixes} that either \(|w|>|v|\) or (\(|w|\leq |v|\) and \(v|_{|w|}\neq w\)). We consider each case separately.

Case 1: Suppose that \(|w|>|v|\). Let \(x: \mathbb{N}\to \{0,1\}\) be defined by \((n)x=1-(|v|)w\) for all \(n\in \mathbb{N}\). Note that \(vx\in v\mathfrak{C}\). Thus \(vx\in w\mathfrak{C}\).
By Lemma~\ref{lem:cone_prefixes}, it follows that \((vx)|_{|w|}=w\).
In particular, \((|v|)w=(|v|)(vx)|_{|w|}=(|v|)vx=(|v|-|v|)x=1-(|v|)w\). 
This is a contradiction.

Case 2: Suppose that \(|w|\leq |v|\) and \(v|_{|w|}\neq w\). Let \(x:\mathbb{N}\to \{0,1\}\) be defined by \((n)x=0\) for all \(n\in \mathbb{N}\). Then \(vx\in v\mathfrak{C}\).
It follows that \(vx\in w\mathfrak{C}\). 
By Lemma~\ref{lem:cone_prefixes}, it follows that \((vx)|_{|w|}=w\). 
As \((vx)|_{|w|}=v|_{|w|}\), this is a contradiction.
\end{proof}

\begin{lemma}\label{lem:words=cones}
\uses{lem:comp_cone_vs_word}
    If \(v,w\in \{0,1\}^*\), then \(w= v \iff w\mathfrak{C} = v\mathfrak{C}\).
\end{lemma}
\begin{proof}
    This follows from Lemma~\ref{lem:comp_cone_vs_word} since \(\preceq\) and \(\subseteq\) are partial orders.
\end{proof}

\begin{lemma}\label{lem:intersecting_cones}
\uses{lem:cone_prefixes}
If \(v,w\in \{0,1\}^*\) and \(v\mathfrak{C} \cap w\mathfrak{C} \neq \varnothing\), then either \(v\mathfrak{C} \subseteq w\mathfrak{C}\) or \(w\mathfrak{C} \subseteq v\mathfrak{C}\).
\end{lemma}
\begin{proof}
    If \(v\mathfrak{C} \cap w\mathfrak{C} \neq \varnothing\), then there is some \(x\in v\mathfrak{C} \cap w\mathfrak{C}\). 
    In particular \(v\preceq x\) and \(w\preceq x\).
    By Lemma~\ref{lem:cone_prefixes}, it follows that \(x|_{|v|}=v\) and \(x|_{|w|}=w\).
    We either have \(|v|\geq |w|\) or \(|w|\geq |v|\). We consider each case separately.

    Case 1: Suppose that \(|v|\geq |w|\). Then \(v|_{|w|}=(x|_{|v|})|_{|w|}=x|_{|w|}=w\). By Lemma~\ref{lem:finite_cone_prefixes}, it follows that \(w\leq v\). Thus by Lemma~\ref{lem:cone_prefixes}, we have \(v\mathfrak{C} \subseteq w\mathfrak{C}\).

    Case 2: Suppose that \(|w|\geq |v|\). Then \(w|_{|v|}=(x|_{|w|})|_{|v|}=x|_{|v|}=v\). By Lemma~\ref{lem:finite_cone_prefixes}, it follows that \(v\leq w\). Thus by Lemma~\ref{lem:cone_prefixes}, we have \(w\mathfrak{C} \subseteq v\mathfrak{C}\).
\end{proof}

\begin{lemma}\label{lem:incompwords_vs_disjoint_cones}
    \uses{lem:intersecting_cones}
    If \(v,w\in \{0,1\}^*\), then \(v\) and \(w\) are incomparable if and only if \(v\mathfrak{C} \cap w \mathfrak{C} =\varnothing\).
\end{lemma}
\begin{proof}
    \((\Rightarrow):\) We show the contrapositive.
    Suppose that \(v\mathfrak{C} \cap w \mathfrak{C} \neq \varnothing\). By Lemma~\ref{lem:intersecting_cones}, it follows that \(v\mathfrak{C} \subseteq w\mathfrak{C}\) or \(w\mathfrak{C} \subseteq v\mathfrak{C}\). Thus by Lemma~\ref{lem:comp_cone_vs_word}, it follows that either \(w\preceq v\) or \(v\preceq w\).
    So \(v\) and \(w\) are comparable.

    \((\Leftarrow):\) We show the contrapositive. Suppose that \(v\) and \(w\) are comparable. 
    
    Case 1: If \(v\preceq w\), then by Lemma~\ref{lem:comp_cone_vs_word}, we have \(w\mathfrak{C} \subseteq v\mathfrak{C}\). In particular, \(v\mathfrak{C}\cap w\mathfrak{C}=w\mathfrak{C}\neq \varnothing\).

    Case 2: If \(w\preceq v\), then by Lemma~\ref{lem:comp_cone_vs_word}, we have \(v\mathfrak{C} \subseteq w\mathfrak{C}\). In particular, \(w\mathfrak{C}\cap v\mathfrak{C}=v\mathfrak{C}\neq \varnothing\).
\end{proof}

\begin{lemma}\label{lem:prefix_code_disjoint_cones}
\uses{lem:intersecting_cones, def:prefix_code}
If \(P \subseteq \{0,1\}^*\), then \(P\) is a prefix code if and only if the sets \(v\mathfrak{C}\) and \(w\mathfrak{C}\) for \(v,w\in P\) are pairwise disjoint.
\end{lemma}
\begin{proof}
    By Definition~\ref{def:prefix_code}, \(P\) is a prefix code if and only if all the words \(v,w\in P\) are pairwise incomparable. By Lemma~\ref{lem:intersecting_cones}, this is equivalent to saying that the sets \(v\mathfrak{C}\) and \(w\mathfrak{C}\) for \(v,w\in P\) are pairwise disjoint.
\end{proof}


\begin{definition}%
  \label{def:complete_prefix_code}
  \uses{def:prefix_code}
A complete prefix code is a prefix code \(P\) such that for all  $x\in \mathfrak{C}$, there is $p_x\in P$ with \(x\in p_x\mathfrak{C}\).
\end{definition}
\begin{lemma}\label{lem:prefix_code=cone_parition}
  \uses{lem:prefix_code_disjoint_cones, lem:words=cones,def:complete_prefix_code,lem:incompwords_vs_disjoint_cones}
  A subset \(P\subseteq \{0,1\}^*\) is a complete prefix code if and only if \(\makeset{w\mathfrak{C}}{\(w\in P\)}\) is a partition of \(\mathfrak{C}\).
\end{lemma}
\begin{proof}
    \((\Rightarrow):\) By Lemma~\ref{lem:prefix_code_disjoint_cones}, the members of the set \(\makeset{w\mathfrak{C}}{\(w\in P\)}\) are pairwise disjoint.
    Let \(x:\mathbb{N}\to \{0,1\}\) be defined by \((n)x=0\) for all \(n\in \mathbb{N}\).
    Note that for all \(w\in P\), we have \(wx\in w\mathfrak{C}\). Thus \(\makeset{w\mathfrak{C}}{\(w\in P\)}\) does not contain it empty set.
    It remains to check that the members of \(\makeset{w\mathfrak{C}}{\(w\in P\)}\) cover \(\mathfrak{C}\). This follows from the definition of a complete prefix code (Definition~\ref{def:complete_prefix_code}).

    \((\Leftarrow):\) As the members of \(\makeset{w\mathfrak{C}}{\(w\in P\)}\) cover \(\mathfrak{C}\), we need only check that \(P\) is a prefix code.
    Let \(p,q\in P\) be such that \(p\neq q\).
    We must show that \(p\) and \(q\) are incomparable.
    We have \(p\mathfrak{C},q\mathfrak{C}\in \makeset{w\mathfrak{C}}{\(w\in P\)}\).
    So either \(p\mathfrak{C}=q\mathfrak{C}\) or \(p\mathfrak{C}\cap q\mathfrak{C}=\varnothing\). We consider these cases separately.

    Case 1: Suppose that \(p\mathfrak{C}=q\mathfrak{C}\). Then by Lemma~\ref{lem:words=cones}, it follows that \(p=q\). This is a contradiction.

    Case 2: Suppose that \(p\mathfrak{C}\cap q\mathfrak{C}=\varnothing\). Then Lemma~\ref{lem:incompwords_vs_disjoint_cones} implies that \(p\) and \(q\) are incomparable as required.

\section{Trees}

We wish to view elements of Thompson's group \(V\) as equivalence classes of tree pairs. We must establish some terminology about binary trees in order to do this.
\begin{definition}%
  \label{def:rooted_binary_tree}
\uses{def:prefix_code,def:concat_letter_set}
 A finite rooted binary tree \(T\) is one of:
 \begin{itemize}
     \item \(\{\varepsilon\}\);
     \item a set of the form \(0T_0 \cup 1T_1 \cup \{\varepsilon\}\) where \(T_0\) and \(T_1\) are finite rooted binary trees.
 \end{itemize}
\end{definition}

\begin{definition}%
  \label{def:leaf_of_a_tree}
\uses{def:rooted_binary_tree}
If \(T\) is a finite rooted binary tree, then a leaf of \(T\) is one of
\begin{itemize}
    \item \(\varepsilon\) if \(T=\{\varepsilon\}\);
    \item \(0l\) if \(T=0T_0 \cup T_1 \cup \{\varepsilon\}\) and \(l\) is a leaf of \(T_0\);
    \item \(1l\) if \(T=0T_0 \cup T_1 \cup \{\varepsilon\}\) and \(l\) is a leaf of \(T_1\).
\end{itemize}
\end{definition}

\begin{definition}%
  \label{def:leaf_set_of_a_tree}
\uses{def:leaf_of_a_tree}
If \(T\) is a finite rooted binary tree, then \(\mathcal{L}(T):=\makeset{t\in T}{\(t\) is a leaf of \(T\)}\).
\end{definition}

\begin{definition}%
  \label{def:tree_pair}
\uses{def:leaf_set_of_a_tree}
A tree pair is a triple \((T_1,T_2,\phi)\) where \(T_1\) and \(T_2\) are finite rooted binary trees and \(\phi:\mathcal{L}(T_1)\to \mathcal{L}(T_2)\) is a bijection.
\end{definition}

\begin{definition}\label{def:tree_elementary_expansions}
    \uses{def:rooted_binary_tree}
    If \(T\) is a rooted binary tree and \(l\) is a leaf of \(T\), then we define the \emph{elementary expansion} \(T'\) of \(T\) at \(l\) to be the set:
    \begin{itemize}
        \item \(\{\varepsilon,0,1\}\) if \(l=\varepsilon\);
        \item \(0T_0'\cup 1 T_1 \cup \{\varepsilon\}\) if \(T=0T_0 \cup 1T_1 \cup \{\varepsilon\}\), \(l=0l'\) for a leaf \(l'\) of \(T_0\), and \(T_0'\) is the elementary expansion of \(T_0\) at \(l'\);
  \item \(0T_0\cup 1 T_1' \cup \{\varepsilon\}\) if \(T=0T_0 \cup 1T_1 \cup \{\varepsilon\}\), \(l=1l'\) for a leaf \(l'\) of \(T_1\), and \(T_1'\) is the elementary expansion of \(T_1\) at \(l'\).
    \end{itemize}
\end{definition}

\begin{lemma}\label{lem:tree_to_codes}
\uses{def:leaf_of_a_tree,def:complete_prefix_code,lem:cone_prefixes}
    If \(T\) is a finite rooted binary tree, then \(\mathcal{L}(T)\) is a complete prefix code.
\end{lemma}
\begin{proof}
    We proceed recursively.

    If \(T=\{\varepsilon\}\), then \(\mathcal{L}(T)=\{\varepsilon\}\). As \(\mathcal{L}(T)\) has only one element, it has no pair of incomparable elements. Thus \(\mathcal{L}(T)\) is a prefix code.
    As \(\varepsilon \preceq x\) for all \(x\in \mathfrak{C}\), it follows that \(\mathcal{L}(T)\) is a complete prefix code.

    Next suppose that \(T=0T_0 \cup 1T_1 \cup \{\varepsilon\}\) for some finite rooted binary trees \(T_0\) and \(T_1\). By recursion, the sets \(\mathcal{L}(T_0)\) and \(\mathcal{L}(T_1)\) are complete prefix codes.
    Moreover, by Definition~\ref{def:leaf_of_a_tree}. We have \(\mathcal{L}(T)=0\mathcal{L}(T_0)\cup 1\mathcal{L}(T_1)\). No word in \(0\mathcal{L}(T_0)\) is comparable with any word of \(1\mathcal{L}(T_1)\) as they have distinct first letters. Moreover the elements of \(\mathcal{L}(T_0)\) being incomparable implies that the elements of \(0\mathcal{L}(T_0)\) are incomparable.
    \luna{todo add reference to reinis's rocq lemma for this}
    Similarly, the elements of \(1\mathcal{L}(T_1)\) are incomparable. Thus \(\mathcal{L}(T)\) is a prefix code.
    Let \(x\in \mathfrak{C}\). 
    There are two cases.

    Case 1: Suppose that \((0)x=0\). It follows from Lemma~\ref{lem:cone_prefixes}, that \(x=0x'\) for some \(x'\in \mathfrak{C}\). There is some \(p\in \mathcal{L}(T_0)\) with \(p\leq x'\). It follows that \(0p \leq x\).

    Case 2: \((0)x=1\). It follows from Lemma~\ref{lem:cone_prefixes}, that \(x=1x'\) for some \(x'\in \mathfrak{C}\). There is some \(p\in \mathcal{L}(T_1)\) with \(p\leq x'\). It follows that \(1p \leq x\).
\end{proof}

\begin{lemma}\label{lem:codes_to_trees}
\uses{def:leaf_of_a_tree,def:complete_prefix_code, lem:finite_complete_prefix_codes}
   If \(P\) is a complete prefix code, then there is a finite rooted binary tree \(T\) with \(\mathcal{L}(T)=P\).
\end{lemma}
\begin{proof}
    Suppose that \(P\) is a complete prefix code.
    By Lemma~\ref{lem:finite_complete_prefix_codes}, \(P\) is a finite set.
    We proceed by induction on the length of the longest element of \(P\).

    Base case: Suppose that \(w\in P\) has maximal length and \(|w|=0\). 
    It follows that \(w=\varepsilon\). As \(P\) is a prefix code, it follows that \(P=\{\varepsilon\}\).
    In particular \(P=\mathcal{L}(T)\) where \(T=\{\varepsilon\}\).

    Inductive hypothesis: Suppose that \(n\in \mathbb{N}\) and for all complete prefix codes \(P\) all of whose words have length at most \(n\), there is a finite rooted binary tree with \(P=\mathcal{L}(T)\).

    Inductive step: Suppose that \(P\) is a complete prefix code and the maximum length of an element of \(P\) is \(n+1\).
    Let \(P_0= P\cap 0\{0,1\}^*\) and  \(P_1= P\cap 1\{0,1\}^*\). 
    For all \(x\in 1 \mathfrak{C}\), there is a prefix of \(x\) belonging to \(P\). 
    As \(P\) is a prefix code and contains a word of positive length, it follows that \(\varepsilon \not\in P\).
    It follows that for all \(x\in 1 \mathfrak{C}\), there is a prefix of \(x\) belonging to \(P_1\).
    Similarly for all \(x\in 0 \mathfrak{C}\), there is a prefix of \(x\) belonging to \(P_0\).
    Let \(P_0'\) be the set obtained from \(P_0\) by removing the first letter of every word. 
    By reinis's thing, the set \(P_0\) is a prefix code. Moreover the elements of \(P_0\) have length at most \(n\).
    As for all \(x\in 0 \mathfrak{C}\), there is a prefix of \(x\) belonging to \(P_0\). Thus \(P_0'\) is a complete prefix code.
    Define a complete prefix code \(P_1'\) similarly.
    By the inductive hypothesis, there are finite rooted binary trees \(T_0\) and \(T_1\) such that
    \[P_0'=\mathcal{L}(T_0)\text{ and }P_1'=\mathcal{L}(T_1).\]
    Let \(T=0T_0\cup 1T_1 \cup \{\varepsilon\}\). Note that by Definition~\ref{def:rooted_binary_tree}, it follows that \(T\) is a finite rooted binary tree.
    Moreover, by Definition~\ref{def:leaf_of_a_tree}, it follows that \(\mathcal{L}(T)=0\mathcal{L}(T_0)\cup 1\mathcal{L}(T_1)=P_0\cup P_1=P\) as required.
\end{proof}

\begin{theorem}\label{thm:tree=complete_prefix_code}
\uses{lem:tree_to_codes,lem:codes_to_trees}
    A subset \(P\subseteq \{0,1\}^*\) is a complete prefix code if and only if it is the set of leafs of a finite rooted binary tree.
\end{theorem}
\begin{proof}
    \((\Rightarrow):\) Lemma~\ref{lem:codes_to_trees}.

    \((\Leftarrow):\) Lemma~\ref{lem:tree_to_codes}.
\end{proof}
